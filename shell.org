* How to read ini config file

#+BEGIN_SRC sh
GetKey(){
    section=$1
    key=$2
    #使用正则表达是获取key所对应的value
    sed -n "/\[$section\]/,/\[.*\]/{
      /^\[.*\]/d
      /^[ \t]*$/d
      /^$/d
      /^#.*$/d
      s/^[ \t]*$key[ \t]*=[ \t]*\(.*\)[ \t]*/\1/p
    }" file_name
}

GetKey $section $key
#+END_SRC
